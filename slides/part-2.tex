			\item \texttt{returnKeyType}
		\end{itemize} \pause
		\item \texttt{Button} \href{https://reactnative.dev/docs/button}{https://reactnative.dev/docs/button}
		\begin{itemize}
			\item \texttt{onPress}
		\end{itemize}
	\end{itemize}
\end{frame}

\begin{frame}{Deleting tasks}
